304  template <typename T>
305  AttribHandle<T> addAttrib( const std::string& name ) {
306  // does the attrib already exist?
307  AttribHandle<T> h = findAttrib<T>( name );
308  if ( isValid( h ) ) return h;
309 
310  // create the attrib
311  smart_pointer_type attrib = std::make_unique<Attrib<T>>( name );
312 
313  // look for a free slot
314  auto it = std::find_if(
315  m_attribs.begin(), m_attribs.end(), []( const auto& attr ) { return !attr; } );
316  if ( it != m_attribs.end() )
317  {
318  it->swap( attrib );
319  h.m_idx = std::distance( m_attribs.begin(), it );
320  }
321  else
322  {
323  m_attribs.push_back( std::move( attrib ) );
324  h.m_idx = m_attribs.size() - 1;
325  }
326  m_attribsIndex[name] = h.m_idx;
327  h.m_name = name;
