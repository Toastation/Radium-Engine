void MyPlugin::registerPlugin( const Ra::Plugins::Context& context ) {
connect( this, &MyPlugin::askForUpdate, &context, &Ra::Plugins::Context::askForUpdate );
connect( this, &MyPlugin::setContinuousUpdate, &context,
